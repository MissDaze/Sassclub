        // ============================================
        // STRIPE CONFIGURATION

        // REPLACE THIS WITH YOUR STRIPE PUBLISHABLE KEY
        // Get it from: https://dashboard.stripe.com/apikeys
        const STRIPE_PUBLISHABLE_KEY = 'pk_test_YOUR_KEY_HERE';
        
        // Initialize Stripe
        const stripe = Stripe(STRIPE_PUBLISHABLE_KEY);
        
        let cartCount = 0;
        
        // Size selection
        document.querySelectorAll('.size-option').forEach(opt => {
            opt.addEventListener('click', function() {
                this.parentElement.querySelectorAll('.size-option').forEach(o => {
                    o.classList.remove('selected');
                });
                this.classList.add('selected');
            });
        });

        // Add to cart / Stripe Checkout
        document.querySelectorAll('.add-to-cart').forEach(btn => {
            btn.addEventListener('click', async function() {
                const productName = this.dataset.product;
                const priceInCents = parseInt(this.dataset.price);
                
                // Get selected size
                const sizeSelector = this.parentElement.querySelector('.size-selector');
                const selectedSize = sizeSelector.querySelector('.size-option.selected').dataset.size;
                
                // Disable button
                this.disabled = true;
                this.textContent = 'Processing...';
                
                try {
                    // Call the API endpoint (same server, so no need for full URL)
                    const response = await fetch('/api/create-checkout-session', {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                        },
                        body: JSON.stringify({
                            product: productName,
                            size: selectedSize,
                            price: priceInCents
                        })
                    });
                    
                    const session = await response.json();
                    
                    // Redirect to Stripe Checkout
                    const result = await stripe.redirectToCheckout({
                        sessionId: session.id
                    });
                    
                    if (result.error) {
                        alert(result.error.message);
                    }
                } catch (error) {
                    console.error('Error:', error);
                    alert('Payment system not yet configured. Please add your Stripe keys!');
                }
                
                // Re-enable button
                this.disabled = false;
                this.textContent = 'Pre-Order Now ðŸ”¥';
            });
        });

        // Filter functionality
        document.querySelectorAll('.filter-tab').forEach(tab => {
            tab.addEventListener('click', function() {
                document.querySelectorAll('.filter-tab').forEach(t => t.classList.remove('active'));
                this.classList.add('active');
                
                const filter = this.dataset.filter;
                document.querySelectorAll('.product-card').forEach(card => {
                    card.style.display = (filter === 'all' || card.dataset.category === filter) ? 'block' : 'none';
                });
            });
        });
    </script>
